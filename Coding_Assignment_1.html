    import numpy as np
from scipy.stats import mode
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import confusion_matrix

def kNN(X_train, Y_train, X_test, k):
    # Compute the squared Euclidean distances between all test points and training points
    dists = np.sqrt(((X_test[:, np.newaxis] - X_train[np.newaxis, :]) ** 2).sum(axis=2))
    
    # Find the indices of the k smallest distances for each test point
    k_nearest_indices = np.argsort(dists, axis=1)[:, :k]
    
    # Extract the labels of the k nearest neighbors
    k_nearest_labels = Y_train[k_nearest_indices]
    
    # Compute the most frequent label among the k nearest neighbors
    predictions, _ = mode(k_nearest_labels, axis=1)
    
    return predictions

    The Bayes rule for binary classification (under the zero-one loss), as derived in class, is:

    predict Y to be 1, if
    P(Y =1|X =x)P(Y =0|X =x)
    , or equivalently
    P (Y = 1 | X = x) 1. P(Y =0|X=x)

    import numpy as np
from scipy.stats import multivariate_normal
from sklearn.metrics import confusion_matrix

def bayes_predict(X_test, m0, m1, s, prior_y0=0.5, prior_y1=0.5):
    # P(Y=0)  P(Y=1) 
    prior_y0 = np.log(prior_y0)
    prior_y1 = np.log(prior_y1)
    
    #  ( s )
    cov = np.eye(X_test.shape[1]) * s**2
    
    #  P(X | Y=0)  P(X | Y=1)
    # log likelihood 
    log_likelihood_y0 = np.zeros(X_test.shape[0])
    log_likelihood_y1 = np.zeros(X_test.shape[0])
    
    #  log likelihood
    for mu0 in m0:
        log_likelihood_y0 += multivariate_normal.logpdf(X_test, mean=mu0, cov=cov)
    
    for mu1 in m1:
        log_likelihood_y1 += multivariate_normal.logpdf(X_test, mean=mu1, cov=cov)
    
    # 
    log_posterior_y0 = log_likelihood_y0 + prior_y0
    log_posterior_y1 = log_likelihood_y1 + prior_y1
    
    #  P(Y=1 | X=x) >= P(Y=0 | X=x) 1 0
    predictions = (log_posterior_y1 >= log_posterior_y0).astype(int)
    
    return predictions
